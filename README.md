## Code Examples

******** RULES ********

(defrule running ""
   (not (run ?))
   (not (sports ?))                                                                                                                     
   => (assert (run-ans (answers "Do you like running? (yes/no)? "))))
   
(defrule contact ""
   (run-ans yes)
   (not (sports ?))                                                                                                                     
   =>
   (assert (contact-ans (answers "Do you prefer contact sports (yes/no)? "))))

(defrule alone ""
   (run-ans no)
   (not (sports ?))   
   =>
   (assert (alone-ans (answers "Do you like to play sport alone (yes/no)? "))))
